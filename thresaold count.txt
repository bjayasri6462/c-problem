#include <stdio.h>

int main() {
      int capacity,n;
      int change;
      int i = 0;
      int occupied = 0;
      int thresholdCount=0;
      int threshold;
      scanf("%d" ,&capacity);
      scanf("%",&n);
      threshold = 0.9*capacity;
      while(i<n){
          scanf("%d",&change);
          occupied +=change;
          if(occupied<0){
              occupied = 0;
          }
          if(occupied>capacity){
              occupied = capacity;
          }
          if(occupied > threshold){
              thresholdCount;
          }
          i++;
      }
      printf("Final occupied Beds:%d\n",occupied);
      printf("Times Exceeded 90%% capacity:%d",thresholdCount);
          
      
      
    

    return 0;
}
