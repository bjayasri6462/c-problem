#include <stdio.h>

int main() {
    int N, original, reversed = 0, digit;

   
    scanf("%d", &N);

    original = N;  
   
    for (; N > 0; N /= 10) {
        digit = N % 10;
        reversed = reversed * 10 + digit;
    }


    if (original == reversed) {
        printf("Yes\n");
    } else {
        printf("No\n");
    }

    return 0;
}
