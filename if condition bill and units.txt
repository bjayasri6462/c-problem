#include <stdio.h>

int main()
{
    int units;
    float bill;

    printf("Enter units: ");
    scanf("%d", &units);

    if(units <= 100){
        bill = units * 5;
    }
    else{
        bill = (100 * 5) + ((units - 100) * 10);
    }

    printf("Bill amount = %.0f", bill);

    return 0;
}
