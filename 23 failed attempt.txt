#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int attempt;
    int i = 1;
    int consecutiveFails = 0;
    int totalFails = 0;
    int lockAttempt = -1;

    while (i <= N) {
        scanf("%d", &attempt);

        if (attempt == 0) {
            totalFails++;
            consecutiveFails++;

            if (consecutiveFails == 3 && lockAttempt == -1) {
                lockAttempt = i;
            }
        } else {
            consecutiveFails = 0;
        }

        i++;
    }

    if (lockAttempt != -1) {
        printf("Lock Triggered At Attempt: %d\n", lockAttempt);
    } else {
        printf("Lock Triggered At Attempt: Not Locked\n");
    }

    printf("Total Failed Attempts: %d\n", totalFails);

    return 0;
}
