#include <stdio.h>

int main() {
    int n, patients;
    int total = 0;
    int overcrowded = 0;

    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        scanf("%d", &patients);
        total += patients;

        if (patients > 100) {
            overcrowded++;
        }
    }

    printf("Total patients: %d\n", total);
    printf("Overcrowded days: %d\n", overcrowded);

    return 0;
}
