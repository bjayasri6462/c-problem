#include <stdio.h>

int main() {
    int atmCash,n;
    scanf("%d",&atmCash);
    scanf("%d",&n);
    int withdrawals[n];
    for(int i =0;i<n;i++){
        scanf("%d",&withdrawals[i]);
        }
        int i=0;
        int successCount = 0;
        while(i<n){
            if(atmCash  >=withdrawals[i]){
                atmCash -= withdrawals[i];
                successCount;
            } else {
                break;
            }
            i++;
        }
            printf("successfull withdrawals:%d\n",successCount);
            printf("Remaining cash :%d\n",atmCash);
           

    return 0;
}
