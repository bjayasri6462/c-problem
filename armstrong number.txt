#include <stdio.h>
#include <math.h>

int main() {
    int N, original, sum = 0, digit, numDigits = 0;

   
    scanf("%d", &N);

    original = N;

   
    int temp = N;
    for (; temp > 0; temp /= 10) {
        numDigits++;
    }

   
    temp = N;
    for (; temp > 0; temp /= 10) {
        digit = temp % 10;
        sum += (int)pow(digit, numDigits);  
    }

   
    if (sum == original) {
        printf("Yes\n");
    } else {
        printf("No\n");
    }

    return 0;
