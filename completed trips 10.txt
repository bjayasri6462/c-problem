#include <stdio.h>

int main() {
    int fuel, n;

    scanf("%d", &fuel);
    scanf("%d", &n);

    int trips[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &trips[i]);
    }

    int completed = 0;

    for (int i = 0; i < n && fuel >= trips[i]; i++) {
        fuel -= trips[i];
        completed++;
    }

    printf("Completed Trips: %d\n", completed);
    printf("Remaining Fuel: %d\n", fuel);

    return 0;
}
