#include <stdio.h>

int main() {
    int n, i = 0;
    int value;
    int max;
    int surgeCount = 0;

    scanf("%d", &n);

    if (n > 0) {
        scanf("%d", &value);   // read first value
        max = value;

        if (value > 5) {
            surgeCount++;
        }

        i = 1;
        while (i < n) {
            scanf("%d", &value);

            if (value > max) {
                max = value;
            }

            if (value > 5) {
                surgeCount++;
            }

            i++;
        }

        printf("Max Usage: %d\n", max);
        printf("Surge Hours: %d\n", surgeCount);
    }

    return 0;
}
