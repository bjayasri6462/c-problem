#include <stdio.h>

int main() {
    int n,i;
    int errors;
    int total = 0;
    int critical = 0;
    

    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        scanf("%d", &errors);
        total += errors;
        
        
        if (errors > 50) {
            critical++;
        }
    }

    printf("Total Errors : %d\n", total);
    printf("Critical Hours: %d\n",critical);

    return 0;
}
