#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int voltage[N];   // declare AFTER reading N
    int min, lowCount = 0;

    for (int i = 0; i < N; i++) {
        scanf("%d", &voltage[i]);
    }

    min = voltage[0];

    for (int i = 0; i < N; i++) {
        if (voltage[i] < min) {
            min = voltage[i];
        }
        if (voltage[i] < 210) {
            lowCount++;
        }
    }

    printf("Minimum voltage: %d\n", min);
    printf("Low voltage events: %d\n", lowCount);

    return 0;
}
