#include <stdio.h>

int main()
{
    int week;
   scanf("%d",&week);
   switch(week){
    case 1:
    printf("Monday");
    break;
    case 2:
     printf("Tuesday");
    break;
    case 3:
    printf("wenesday");
    break;
    case 4 :
    printf("thursday");
    break;
    case 5:
    printf("friday");
    break;
    default:
    printf("default");
    return 0;
   }
    
   }
