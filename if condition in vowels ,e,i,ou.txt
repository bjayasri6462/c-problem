int main()
{
    char ch;
    scanf("%c", &ch);

    switch(ch)
    {
        case 'A':
        case 'E':
        case 'I':
        case 'O':
        case 'U':
        case 'a':
        case 'e':
        case 'i':
        case 'o':
        case 'u':
            printf("vowel");
            break;

        default:
            printf("not vowel");
    }

    return 0;
}
