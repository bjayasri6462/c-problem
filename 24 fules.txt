#include <stdio.h>

int main() {
    int fuel, N;
    scanf("%d", &fuel);
    scanf("%d", &N);

    int burn;
    int i = 1;
    int emergencyStage = -1;
    int fuelWasted = 0;

    while (i <= N) {
        scanf("%d", &burn);
        fuel -= burn;

        if (fuel < 0 && emergencyStage == -1) {
            emergencyStage = i;
            fuelWasted = -fuel;
        }

        i++;
    }

    if (emergencyStage != -1) {
        printf("Emergency Stage: %d\n", emergencyStage);
        printf("Fuel Wasted: %d\n", fuelWasted);
    } else {
        printf("Emergency Stage: Not Detected\n");
        printf("Fuel Wasted: 0\n");
    }

    return 0;
}
