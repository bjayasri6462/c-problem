#include <stdio.h>

int main() {

    int batteryPercent, N;
    int drain, totalDrain = 0;

    scanf("%d %d", &batteryPercent, &N);

    for (int i = 0; i < N; i++) {
        scanf("%d", &drain);
        totalDrain += drain;

        int remaining = batteryPercent - totalDrain;
        if (remaining < 0) {
            remaining = 0;
        }

        printf("Remaining battery: %d\n", remaining);
    }

    return 0;
}
