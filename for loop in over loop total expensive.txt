#include <stdio.h>

int main() {
    int n, i;
    int total = 0;
    int overspend = 0;
    int expense, budget = 100;  // example daily budget

    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        scanf("%d", &expense);   // read daily expense
        total += expense;

        if (expense > budget) {
            overspend++;
        }
    }

    printf("Total expense: %d\n", total);
    printf("Overspend days: %d\n", overspend);

    return 0;
}
